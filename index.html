import tkinter as tk from tkinter import messagebox # --- CONFIGURAÇÕES DE CORES (PALETA VERDE) --- VERDE_FONDO = "#1e3d2f" # Verde escuro para o fundo VERDE_CARD = "#2d5a44" # Verde médio para os blocos VERDE_BOTAO = "#4caf50" # Verde brilhante para botões BRANCO = "#ffffff" TEXTO_VERDE = "#81c784" def calcular_sustentabilidade(): try: total = float(entry_total.get()) app = float(entry_app.get()) pesticidas = var_pesticidas.get() if app > total: messagebox.showerror("Erro", "A área de APP não pode ser maior que a área total!") return # Cálculo do percentual de preservação pct_preservacao = (app / total) * 100 # Lógica de pontuação para o Engenheiro Agrônomo pontos = 0 if pct_preservacao >= 20: pontos += 50 elif pct_preservacao >= 10: pontos += 25 if pesticidas == "Baixo/Biológico": pontos += 50 elif pesticidas == "Moderado": pontos += 25 # Resultado resultado_txt = f"Área Preservada: {pct_preservacao:.1f}%\n" if pontos >= 80: resultado_txt += "Selo: 🟢 Forte & Sustentável (Excelente)" elif pontos >= 50: resultado_txt += "Selo: 🟡 Em Transição Sustentável" else: resultado_txt += "Selo: 🔴 Alerta Necessário (Ajustar Manejo)" lbl_resultado.config(text=resultado_txt, fg=BRANCO) except ValueError: messagebox.showerror("Erro de Digitação", "Por favor, insira números válidos para as áreas.") # --- CONSTRUÇÃO DA INTERFACE VISUAL --- root = tk.Tk() root.title("Agrinho: Agro Forte, Futuro Sustentável") root.geometry("450x550") root.configure(bg=VERDE_FONDO) # Cabeçalho lbl_titulo = tk.Label(root, text="PAINEL DO AGRÔNOMO", font=("Arial", 16, "bold"), bg=VERDE_FONDO, fg=TEXTO_VERDE) lbl_titulo.pack(pady=20) # Card de Entrada (Área Total) frame_dados = tk.Frame(root, bg=VERDE_CARD, bd=2, relief="groove") frame_dados.pack(pady=10, padx=20, fill="both", expand=True) tk.Label(frame_dados, text="Área Total da Propriedade (ha):", bg=VERDE_CARD, fg=BRANCO, font=("Arial", 11)).pack(pady=5) entry_total = tk.Entry(frame_dados, font=("Arial", 11), justify="center") entry_total.pack(pady=5) # Card de Entrada (APP) tk.Label(frame_dados, text="Área de Preservação / APP (ha):", bg=VERDE_CARD, fg=BRANCO, font=("Arial", 11)).pack(pady=5) entry_app = tk.Entry(frame_dados, font=("Arial", 11), justify="center") entry_app.pack(pady=5) # Dropdown de Pesticidas tk.Label(frame_dados, text="Nível de Defensivos Químicos:", bg=VERDE_CARD, fg=BRANCO, font=("Arial", 11)).pack(pady=5) var_pesticidas = tk.StringVar(value="Baixo/Biológico") opt_pesticidas = tk.OptionMenu(frame_dados, var_pesticidas, "Baixo/Biológico", "Moderado", "Alto/Intensivo") opt_pesticidas.config(font=("Arial", 10), bg=VERDE_FONDO, fg=BRANCO) opt_pesticidas.pack(pady=5) # Botão de Ação btn_analisar = tk.Button(root, text="GERAR DIAGNÓSTICO", font=("Arial", 11, "bold"), bg=VERDE_BOTAO, fg=BRANCO, command=calcular_sustentabilidade) btn_analisar.pack(pady=20) # Painel de Resultado lbl_resultado = tk.Label(root, text="Insira os dados acima para calcular.", font=("Arial", 12, "bold"), bg=VERDE_FONDO, fg=TEXTO_VERDE, justify="center") lbl_resultado.pack(pady=10) root.mainloop() #
